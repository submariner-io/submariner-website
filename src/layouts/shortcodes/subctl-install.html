Download the subctl binary and make it available on your PATH. 

mkdir -p ~/.local/bin
curl -L -o ~/.local/bin/subctl https://github.com/submariner-io/submariner-operator/releases/download/v0.3.0/subctl-v0.3.0-linux-amd64
chmod a+x ~/.local/bin/subctl
export PATH=$PATH:~/.local/bin
echo export PATH=$PATH:~/.local/bin >> ~/.profile
